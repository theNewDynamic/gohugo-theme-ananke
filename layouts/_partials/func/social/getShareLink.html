{{- $context := .context -}} {{- $setup := .setup -}} {{- $separator := "&" -}} {{- $title := $context.Title | htmlEscape -}} {{- $description := $context.Summary | plainify | htmlEscape -}} {{- $permalink := $context.Permalink | htmlEscape -}} {{- with $setup.separator -}} {{- $separator = . -}} {{- end -}} {{- $link := printf "%s%s" $setup.link "?" -}} {{- range $key, $value := $setup.particles -}} {{- if eq $key "params" -}} {{- $link = printf "%s%s%s" $link $separator $value -}} {{- else -}} {{- if eq $value "description" -}} {{- $link = printf "%s%s%s" $link $separator (querify $key $description) -}} {{- else if eq $value "title" -}} {{- $link = printf "%s%s%s" $link $separator (querify $key $title) -}} {{- else if eq $value "permalink" -}} {{- $link = printf "%s%s%s" $link $separator (querify $key $permalink) -}} {{- end -}} {{- end -}} {{- end -}} {{- return $link -}}
